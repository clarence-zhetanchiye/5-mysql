示例二、
    CREATE EVENT IF NOT EXISTS my_event
    ON SCHEDULE EVERY 1 MONTH  STARTS  DATE_ADD(CURDATE(), INTERVAL 1 MONTH)
    ON COMPLETION PRESERVE
    ENABLE
    DO CALL my_processor();
